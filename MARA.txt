MALZEME:
LOAD
replace(ltrim(replace(MATNR,'0',' ')),' ','0') AS MALZEME,  
MTART AS MALZEME_TURU,
MATKL AS MAL_GRUBU,
HOEHE AS YUKSEKLIK,
BREIT AS GENISLIK,
LAENG AS UZUNLUK
;

SQL SELECT MATNR, MTART, MATKL, HOEHE, BREIT, LAENG 
FROM MARA
WHERE LVORM = ' '
;

Left Join(MALZEME)
LOAD
replace(ltrim(replace(MATNR,'0',' ')),' ','0')  AS MALZEME,     
MAKTX AS MALZEME_METNI
;

SQL SELECT MATNR, MAKTX 
FROM MAKT 
WHERE SPRAS = 'TR'
;

Left Join (MALZEME)
LOAD

MATKL AS MAL_GRUBU,
WGBEZ AS MG_TANIMI
;
