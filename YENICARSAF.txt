FI_Rapor:

LOAD
BELNR AS BELGENO,
BLART AS BELGETIPI,
BLDAT AS BELGETARIHI,
GJAHR AS MALIYIL,
WAERS AS PARABIRIMI,
XBLNR AS REFERANS,
BUKRS AS SIRKETKODU,
BUDAT AS KAYITTARIH
;

SQL SELECT BELNR, BLART, BLDAT, GJAHR, WAERS, XBLNR, BUKRS, BUDAT FROM BKPF


   WHERE BELNR IN (
    '1100000612'
)
AND GJAHR = '2025'
AND BUKRS = '2000'
;

Left Join(FI_Rapor)

LOAD
BELNR AS BELGENO,
GJAHR AS MALIYIL,
MWSKZ AS VERGITURU,
BSCHL AS KAYITANAHTARI,
BUZEI AS OGESAYISI,
WRBTR AS TUTAR,
HKONT AS DENEME1,
BUKRS AS SIRKETKODU
;

SQL SELECT BELNR, MWSKZ, BSCHL, BUZEI, GJAHR, WRBTR, HKONT, BUKRS FROM BSEG

    WHERE BELNR IN (
    '1100000612'
)
AND GJAHR = '2025'
AND BUKRS = '2000'
;

Left Join(FI_Rapor)

LOAD
    SAKNR AS DENEME1,       // HKONT ile eşleştirilecek hesap numarası
    TXT20 AS HESAP_ADI      // Hesap tanımı
;

SQL SELECT SAKNR, TXT20
FROM SKAT
WHERE SPRAS = 'TR'  // veya 'TR' Türkçe ise
;
