LET vStartDate = Date(MonthStart(Today()), 'YYYYMMDD');
LET vEndDate   = Date(MonthEnd(Today()),   'YYYYMMDD');
KONTROLAKARAOGLAN:
LOAD
BELNR AS BELGE_NO,
BUKRS AS SIRKET_KODU,
GJAHR AS YIL,
BLDAT AS BELGE_TARIHI,
BLART AS BELGE_TURU,
USNAM AS OLUSTURAN_KULLANICI,
TCODE AS TRANSAKSIYON_KODU,
AEDAT AS DEGISIKLIK_TARIHI,
XBLNR AS REFERANS
WHERE Len(Trim(XBLNR)) > 8
;

SQL SELECT 
    BELNR,
    BUKRS,
    GJAHR,
    BLDAT,
    BLART,
    USNAM,
    TCODE,
    AEDAT,
    XBLNR
FROM BKPF
WHERE USNAM = 'MGUL'
AND BUKRS = '2000'
AND BLDAT BETWEEN '$(vStartDate)' AND '$(vEndDate)'
AND BLART IN ('KR', 'Y1')
